// SPDX-License-Identifier: MIT
pragma solidity >=0.4.25 <0.7.0;
 
import "./utilities/Ownable.sol";
import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/utils/Counters.sol";
 
/*
 * @dev Implementation of a multi-token for Election.sol and ElectionFactory.sol.
 * Every election contract has unique tokens.
 */
contract VoteToken is ERC1155, Ownable {
    // Counter for identification of election
    using Counters for Counters.Counter;
    Counters.Counter private _electionCounter;
 
    // Struct for ElectionContract information
    struct ElectionContract {
        uint256 idElection;
        bool isActive;
        address[] requestors;
        mapping(address => bool) alreadyRequested;
    }
 
    // Mapping from address to ElectionContract
    mapping(address => ElectionContract) private _contracts;
 
    // Event triggered by getting token
    event TokenEvent(bool _isTokenAvailable);
 
    /**
     * @dev Constructor inherits from ERC1155
     * takes URI as identifier for Tokens
     */
    constructor() public ERC1155("FungibleVoteToken/{id}") {}
 
    /**
     * @dev checks if person already voted
     */
    modifier onlyNewRequests(address _election) {
        Erequire(
            _contracts[_election].isActive &&
                !_contracts[_election].alreadyRequested[msg.sender],
            "VoteToken already claimed"
        );
        _;
    }
 
    /**
     * @dev adds an election struct
     *
     * Requirements:
     *
     * - only for owner
     * - `_election` does not exist
     */
    function addElection(address _election) public onlyOwner {
        Erequire(!_contracts[_election].isActive, "Election already exists");
        _electionCounter.increment();
        uint256 newId = _electionCounter.current();
        _contracts[_election].idElection = newId;
        _contracts[_election].isActive = true;
    }
 
    /**
     * @dev deletes an election struct
     * already existing tokens of this election are not burned
     *
     * Requirements:
     *
     * - only for owner
     * - `_election` exists
     */
    function delElection(address _election) public onlyOwner {
        Erequire(_contracts[_election].isActive, "Election does not exist");
        for (uint256 i = 0; i < _contracts[_election].requestors.length; ++i) {
            delete _contracts[_election].alreadyRequested[
                _contracts[_election].requestors[i]
            ];
        }
        delete _contracts[_election];
    }
 
    /**
     * @dev adds the requestor to an election struct
     *
     * Requirements:
     *
     * - `_election` exists
     * - request of sender is new
     */
    function setRequestor(address _election, address _requestor) private
        onlyNewRequests(_election)
    {
        Erequire(_contracts[_election].isActive, "Election does not exist");
        _contracts[_election].alreadyRequested[_requestor] = true;
        _contracts[_election].requestors.push(_requestor);
    }
 
    /**
     * @dev mints VoteToken and send to msg.sender
     *
     * Emits a {TransferSingle} and {tokenEvent} event.
     *
     * Requirements:
     *
     * - `_election` exists
     * - request of sender is new
     */
    function getVoteToken(address _election) public {
        setRequestor(_election, msg.sender);
        _mint(msg.sender, _contracts[_election].idElection, 1, "");
        emit TokenEvent(true);
    }
 
    /**
     * @dev mints VoteToken and delegate to _receiver
     *
     * Emits a {TransferSingle} and {tokenEvent} event.
     *
     * Requirements:
     *
     * - `_election` exists
     * - request of sender is new
     */
    function delegateVoteToken(address _election, address _receiver) public {
        setRequestor(_election, msg.sender);
        _mint(_receiver, _contracts[_election].idElection, 1, "");
        emit TokenEvent(true);
    }
 
    /**
     * @dev gets id of election
     *
     * Requirements:
     *
     * - `_election` exists
     */
    function getID(address _election) public view returns (uint256) {
        require(_contracts[_election].isActive, "Election does not exist");
        return _contracts[_election].idElection;
    }
}
